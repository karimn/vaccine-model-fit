plot_model_param <- function(optim_data, .baseline_param = baseline_param, other_types = NULL) {
  baseline_param_data <- .baseline_param %>% 
    enframe(name = "param_name", value = "param_val")
  
  optim_data %>% 
    select(run_id, month, param_data, {{ other_types }}) %>% 
    unnest(param_data) %>% 
    group_by(run_id, month, {{ other_types }}) %>%
    filter(step == n()) %>%
    ungroup() %>%
    select(!c(obj, summary)) %>%  
    pivot_longer(., -c(step, run_id, month, {{ other_types }}), names_to = "param_name", values_to = "param_val") %>% 
    ggplot() +
    geom_hline(aes(yintercept = param_val), linetype = "dashed", 
               data = . %>% 
                 modelr::data_grid(param_name, month) %>% 
                 left_join(baseline_param_data, by = "param_name")) +
    geom_line(aes(month, param_val, group = str_c(run_id, {{ other_types }}), color = {{ other_types }}), alpha = 0.5) +
    scale_x_date("", breaks = "2 months", labels = scales::date_format("%m/%y")) +
    labs(
      title = "Parameter Fit", 
      y = "",
      caption = "Dashed horizontal line: baseline AHT model parameters."
    ) +
    facet_wrap(vars(param_name)) +
    theme_minimal() +
    theme(axis.text.x = element_text(angle = 45, hjust = 1)) +
    NULL
}

plot_success_rates <- function(optim_data, .baseline_model_success_rates = baseline_model_success_rates, other_types = NULL) {
  optim_data %>% 
    ggplot(aes(month, success_rate)) +
    geom_hline(aes(yintercept = success_rate), linetype = "dashed",
               data = .baseline_model_success_rates %>% 
                 distinct(cgd_vaccine_id, .keep_all = TRUE)) +
    geom_line(
      data = . %>%
        filter(run_id == 1) %>%
        select(month, cgd_summary) %>%
        rowwise() %>%
        mutate(cgd_summary = list(cgd_summary$success_rates)) %>%
        ungroup() %>%
        unnest(cgd_summary) %>% 
        left_join(cgd_id_dict, by = c("vacc_group_id" = "candInd")) %>% 
        arrange(cgd_vaccine_id)
    ) +
    geom_line(
      aes(group = str_c(run_id, {{ other_types }}), color = {{ other_types }}), alpha = 0.5,
      data = . %>%
        rowwise() %>%
        mutate(run_summary = list(semi_join(run_summary$success_rates, cgd_summary$success_rates, by = "vacc_group_id"))) %>%
        ungroup() %>%
        select(run_id, month, run_summary, {{ other_types }}) %>%
        unnest(run_summary) %>% 
        left_join(cgd_id_dict, by = c("vacc_group_id" = "candInd")) %>% 
        arrange(cgd_vaccine_id)
    ) +
    scale_x_date("", breaks = "2 months", labels = scales::date_format("%m/%y")) +
    labs(title = "Success Probability By Vaccine Candidate",
         subtitle = "Restricted to phase 2 and 3 vaccines.",
         caption = "Using 10 runs.
                    Solid line: the moment calculated from the CGD model data.
                    Dashed horizontal line: baseline AHT model parameters.",
         y = ""
       ) +
    facet_wrap(
      vars(cgd_vaccine_id),
      labeller = labeller(
        cgd_vaccine_id = function(ids) {
          left_join(tibble(cgd_vaccine_id = as.integer(ids)), cgd_id_dict, by = c("cgd_vaccine_id")) %$%
            str_glue("CGD ID: {cgd_vaccine_id}\n{Platform}\n{Subcategory}\n{phase}") %>%
            as.character()
        }
      )
    ) +
    theme_minimal() +
    theme(axis.text.x = element_text(angle = 45, hjust = 1)) +
    NULL
}

plot_vacc_type_success_rates <- function(optim_data, .baseline_model_success_rates = baseline_model_success_rates, other_types = NULL) {
  optim_data %>% 
    ggplot(aes(month, success_rate)) +
    geom_hline(aes(yintercept = success_rate), linetype = "dashed",
               data = .baseline_model_success_rates %>% 
                 group_by(vacc_type_id) %>% 
                 summarize(success_rate = mean(success_rate), .groups = "drop")) +
    geom_line(
      aes(group = str_c(run_id, {{ other_types }}), color = {{ other_types }}), alpha = 0.5,
      data = . %>%
        rowwise() %>%
        mutate(run_summary = list(semi_join(run_summary$success_rates, cgd_summary$success_rates, by = "vacc_group_id"))) %>%
        ungroup() %>%
        select(run_id, month, run_summary, {{ other_types }}) %>%
        unnest(run_summary) %>% 
        left_join(cgd_id_dict, by = c("vacc_group_id" = "candInd")) %>% 
        distinct(run_id, month, vacc_type_id, {{ other_types }}, .keep_all = TRUE)
    ) +
    geom_line(
      data = . %>%
        filter(run_id == 1) %>%
        select(month, cgd_summary) %>%
        rowwise() %>%
        mutate(cgd_summary = list(cgd_summary$success_rates)) %>%
        ungroup() %>%
        unnest(cgd_summary) %>% 
        left_join(cgd_id_dict, by = c("vacc_group_id" = "candInd")) %>% 
        group_by(month, vacc_type_id) %>% 
        summarize(success_rate = mean(success_rate), .groups = "drop")
    ) +
    scale_x_date("", breaks = "2 months", labels = scales::date_format("%m/%y")) +
    labs(title = "Success Probability By Platform-Subcategory-Phase",
         subtitle = "Restricted to phase 2 and 3 vaccines.",
         caption = "Using 10 runs.
                    Solid line: the moment calculated from the CGD model data.
                    Dashed horizontal line: baseline AHT model parameters.",
         y = ""
    ) +
    facet_wrap(
      vars(vacc_type_id),
      labeller = labeller(
        vacc_type_id = function(ids) {
          left_join(tibble(vacc_type_id = as.integer(ids)), 
                    distinct(cgd_id_dict, vacc_type_id, Platform, Subcategory, phase), 
                    by = c("vacc_type_id")) %$%
            str_glue("{Platform}\n{Subcategory}\n{phase}") %>%
            as.character()
        }
      )
    ) +
    theme_minimal() +
    theme(axis.text.x = element_text(angle = 45, hjust = 1)) +
    NULL
}
